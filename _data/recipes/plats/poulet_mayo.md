## Instructions

    1 kg poulet  
    1 oignon  
    1 piment  
    4 gousses d’ail  
    1 tbsp gingembre  
    1 tbsp persil  
    5 tbsp mayonnaise  
    1 poivron vert  
 
    assaisonnement vert := écraser(
        poivron vert, 
        persil, 
        gousses d’ail, 
        gingembre, 
        oignon,
        piment
    )

    poulet assaisonné = macérer 1h (
        débiter en morceaux(poulet),
        assaisonnement vert,
        sel,
        poivre,
        paprika
    )
        
    poulet grillé, oignons brûlés = griller au four (
        "poulet assaisonné (40 min)",
        oignons coupés en deux 
    )

    griller au four 15min (
        empaqueter en papillotes (
            poulet grillé,
            mayonnaise,
            oignons brûlés
        )
    )
